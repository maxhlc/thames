41  template<class T>
42  std::vector<T> cartesian_to_keplerian(const std::vector<T>& RV, const T& mu){
43  // Extract position and velocity vectors
44  std::vector<T> R = {RV[0], RV[1], RV[2]};
45  std::vector<T> V = {RV[3], RV[4], RV[5]};
46 
47  // Declare units vectors
48  const std::vector<T> K = {0.0, 0.0, 1.0};
49 
50  // Set constants
51  const double atol = 1e-12;
52 
53  // Calculate state magnitudes
54  T r = thames::vector::geometry::norm3(R);
55  T v = thames::vector::geometry::norm3(V);
56 
57  // Calculate semi-major axis
58  T sma = 1.0/(2.0/r - pow(v, 2.0)/mu);
59 
60  // Calculate angular momentum vector and magnitude
61  std::vector<T> H = thames::vector::geometry::cross3(R, V);
62  T h = thames::vector::geometry::norm3(H);
63 
64  // Calculate eccentricity vector and magnitude
65  std::vector<T> E = thames::vector::geometry::cross3(V, H)/mu - R/r;
66  T e = thames::vector::geometry::norm3(E);
67 
68  // Calculate inclination
69  T inc = acos(H[2]/h);
70 
71  // Check for circular and equatorial orbits
72  bool e_near = fabs(e) < atol;
73  bool inc_near = fabs(inc) < atol;
74 
75  // Calculate right ascension of the ascending node
76  std::vector<T> N = thames::vector::geometry::cross3(K, H);
77  T n = thames::vector::geometry::norm3(N);
78  T raan;
79  if (inc_near) {
80  raan = 0.0;
81  } else {
82  raan = acos(N[0]/n);
83  if(N[1] < 0.0){
84  raan = 2.0*M_PI - raan;
85  }
86  }
87 
88  // Calculate argument of periapsis
89  T aop;
90  if (inc_near & e_near) {
91  aop = 0.0;
92  } else if (inc_near) {
93  aop = atan2(E[1], E[0]);
94  if(H[2] < 0.0){
95  aop = 2.0*M_PI - aop;
96  }
97  } else {
98  aop = acos(thames::vector::geometry::dot3(N, E)/(n*e));
99  if(E[2] < 0.0){
100  aop = 2.0*M_PI - aop;
101  }
102  }
103 
104  // Calculate true anomaly
105  T ta;
106  if (inc_near & e_near) {
107  ta = acos(R[0]/r);
108  if (V[0] > 0.0) {
109  ta = 2.0*M_PI - ta;
110  }
111  } else if (e_near) {
112  ta = acos(thames::vector::geometry::dot3(N, R)/(n*r));
113  if (R[2] < 0.0) {
114  ta = 2.0*M_PI - ta;
115  }
116  } else {
117  ta = acos(thames::vector::geometry::dot3(E, R)/(e*r));
118  if (thames::vector::geometry::dot3(R, V) < 0.0) {
119  ta = 2.0*M_PI - ta;
120  }
121  }
122 
123  // Store elements in the Keplerian elements vector
124  std::vector<T> keplerian = {
125  sma,
126  e,
127  inc,
128  raan,
129  aop,
130  ta
131  };
132 
133  // Return Keplerian elements vector
134  return keplerian;
135  }

138  template<class T>
139  std::vector<T> keplerian_to_cartesian(const std::vector<T>& keplerian, const T& mu){
140  // Extract Keplerian elements
141  T sma = keplerian[0];
142  T e = keplerian[1];
143  T inc = keplerian[2];
144  T raan = keplerian[3];
145  T aop = keplerian[4];
146  T ta = keplerian[5];
147 
148  // Calculate angle trigs
149  T cinc = cos(inc), sinc = sin(inc);
150  T craan = cos(raan), sraan = sin(raan);
151  T caop = cos(aop), saop = sin(aop);
152  T cta = cos(ta), sta = sin(ta);
153 
154  // Calculate eccentric anomaly
155  T E = 2.0*atan(sqrt((1.0 - e)/(1.0 + e))*tan(ta/2.0));
156 
157  // Calculate radial distance
158  T r = sma*(1.0 - pow(e, 2.0))/(1.0 + e*cta);
159 
160  // Calculate position and velocity vectors in the orbital frame
161  std::vector<T> o = {
162  r*cta,
163  r*sta,
164  0.0
165  };
166 
167  T fac = sqrt(mu*sma)/r;
168  std::vector<T> dodt = {
169  -fac*sin(E),
170  fac*sqrt(1.0 - pow(e, 2.0))*cos(E),
171  0.0
172  };
173 
174  // Calculate rotation angles
175  T ang[3][2];
176  ang[0][0] = caop*craan - saop*cinc*sraan;
177  ang[0][1] = -saop*craan - caop*cinc*sraan;
178  ang[1][0] = caop*sraan + saop*cinc*craan;
179  ang[1][1] = caop*cinc*craan - saop*sraan;
180  ang[2][0] = saop*sinc;
181  ang[2][1] = caop*sinc;
182 
183  // Transform the position and velocity vectors to the inertial frame
184  std::vector<T> RV(6);
185  for(unsigned int ii=0; ii<3; ii++){
186  RV[ii] = o[0]*ang[ii][0] + o[1]*ang[ii][1];
187  RV[ii+3] = dodt[0]*ang[ii][0] + dodt[1]*ang[ii][1];
188  }
189 
190  // Return Cartesian state
191  return RV;
192  }
