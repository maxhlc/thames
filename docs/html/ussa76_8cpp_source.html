50  template<class T>
51  T USSA76AtmosphereModel<T>::density(T alt) const {
52  // Declare index variable
53  std::size_t ii;
54 
55  // Handle altitudes outside of the range
56  T altselect = alt;
57  if (altselect > m_geo.back()) {
58  altselect = m_geo.back();
59  } else if (altselect < m_geo.front()) {
60  altselect = m_geo.front();
61  }
62 
63  // Determine interpolation interval
64  for (std::size_t jj = 0; jj < m_geo.size() - 1; jj++) {
65  if (altselect >= m_geo[jj] && altselect < m_geo[jj+1])
66  ii = jj;
67  }
68  if (altselect >= m_geo.back())
69  ii = m_geo.size() - 1;
70 
71  // Exponential interpolation
72  T rho = m_rho[ii]*std::exp(-(alt - m_geo[ii])/m_scale[ii]);
73 
74  // Return density
75  return rho;
76  }
