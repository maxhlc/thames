53  template<class T>
54  std::vector<T> convert_state(const T& t, const std::vector<T>& state, const T& mu, const StateTypes& statetype1, const StateTypes& statetype2, const std::shared_ptr<const BasePerturbation<T>> perturbation) {
55  // Return input directly if the two types match
56  if (statetype1 == statetype2)
57  return state;
58 
59  // Cartesian -> GEqOE
60  if (statetype1 == CARTESIAN && statetype2 == GEQOE)
61  return thames::conversions::geqoe::cartesian_to_geqoe(t, state, mu, perturbation);
62 
63  // GEqOE -> Cartesian
64  if (statetype1 == GEQOE && statetype2 == CARTESIAN)
65  return thames::conversions::geqoe::geqoe_to_cartesian(t, state, mu, perturbation);
66 
67  // Cartesian -> Keplerian
68  if (statetype1 == CARTESIAN && statetype2 == KEPLERIAN)
69  return thames::conversions::keplerian::cartesian_to_keplerian(state, mu);
70 
71  // Keplerian -> Cartesian
72  if (statetype1 == KEPLERIAN && statetype2 == CARTESIAN)
73  return thames::conversions::keplerian::keplerian_to_cartesian(state, mu);
74 
75  // Throw error if combinations not accepted
76  throw std::runtime_error("Unsupported state conversion");
77  }
