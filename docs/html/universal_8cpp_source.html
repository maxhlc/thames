52  template<class T>
53  std::array<T, 6> convert_state(const T& t, const std::array<T, 6>& state, const T& mu, const StateTypes& statetype1, const StateTypes& statetype2, const std::shared_ptr<const BasePerturbation<T>> perturbation) {
54  // Return input directly if the two types match
55  if (statetype1 == statetype2)
56  return state;
57 
58  // Cartesian -> GEqOE
59  if (statetype1 == CARTESIAN && statetype2 == GEQOE)
60  return thames::conversions::geqoe::cartesian_to_geqoe(t, state, mu, perturbation);
61 
62  // GEqOE -> Cartesian
63  if (statetype1 == GEQOE && statetype2 == CARTESIAN)
64  return thames::conversions::geqoe::geqoe_to_cartesian(t, state, mu, perturbation);
65 
66  // Cartesian -> Keplerian
67  if (statetype1 == CARTESIAN && statetype2 == KEPLERIAN)
68  return thames::conversions::keplerian::cartesian_to_keplerian(state, mu);
69 
70  // Keplerian -> Cartesian
71  if (statetype1 == KEPLERIAN && statetype2 == CARTESIAN)
72  return thames::conversions::keplerian::keplerian_to_cartesian(state, mu);
73 
74  // Throw error if combinations not accepted
75  throw std::runtime_error("Unsupported state conversion");
76  }
