62  template<class T>
63  std::array<T, 3> Drag<T>::acceleration_nonpotential(const T& t, const std::array<T, 3>& R, const std::array<T, 3>& V) const {
64  // Calculate factors
65  T radius = (m_isNonDimensional) ? m_radius/(m_factors->length) : m_radius;
66  T w = (m_isNonDimensional) ? m_w*(m_factors->time) : m_w;
67  T Cd = m_Cd;
68  T A = (m_isNonDimensional) ? m_A/std::pow(m_factors->length, 2) : m_A;
69 
70  // Calculate altitude
71  T r = thames::vector::geometry::norm3(R);
72  T alt = r - radius;
73 
74  // Calculate atmospheric density (including conversion to kg/km^3)
75  if (m_isNonDimensional)
76  alt *= m_factors->length;
77  T rho = m_model.density(alt) * 1e9;
78 
79  // Calculate factors which include mass (cancelled via rho/mass) and non-dimensionalise as required
80  T massfac = rho/m_m;
81  if (m_isNonDimensional)
82  massfac *= std::pow(m_factors->length, 3);
83 
84  // Calculate velocity relative to the atmosphere
85  std::array<T, 3> W = {0, 0, w};
86  std::array<T, 3> Vrel = V - thames::vector::geometry::cross3(W, R);
87  T vrel = thames::vector::geometry::norm3(Vrel);
88  std::array<T, 3> uv = Vrel/vrel;
89 
90  // Calculate acceleration due to drag
91  std::array<T, 3> Ad = -0.5*Cd*A*massfac*pow(vrel, 2)*uv;
92 
93  // Return acceleration
94  return Ad;
95  }
