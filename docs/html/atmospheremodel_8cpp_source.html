88  template<class T>
89  T AtmosphereModel<T>::density(T alt) const {
90  // Declare index variable
91  std::size_t ii;
92 
93  // Handle altitudes outside of the range
94  T altselect = alt;
95  if (altselect > m_geo.back()) {
96  altselect = m_geo.back();
97  } else if (altselect < m_geo.front()) {
98  altselect = m_geo.front();
99  }
100 
101  // Determine interpolation interval
102  for (std::size_t jj = 0; jj < m_geo.size() - 1; jj++) {
103  if (altselect >= m_geo[jj] && altselect < m_geo[jj+1])
104  ii = jj;
105  }
106  if (altselect >= m_geo.back())
107  ii = m_geo.size() - 1;
108 
109  // Exponential interpolation
110  T rho = m_rho[ii]*std::exp(-(alt - m_geo[ii])/m_scale[ii]);
111 
112  // Return density
113  return rho;
114  }
