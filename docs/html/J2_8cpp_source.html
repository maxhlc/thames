56  template <class T>
57  std::vector<T> J2<T>::acceleration_total(const T& t, const std::vector<T>& R, const std::vector<T>& V) const {
58  // Calculate factors
59  const T mu = (m_isNonDimensional) ? m_mu/m_factors->grav : m_mu;
60  const T J2 = m_J2;
61  const T radius = (m_isNonDimensional) ? m_radius/m_factors->length : m_radius;
62 
63  // Extract position components
64  const T x = R[0], y = R[1], z = R[2];
65 
66  // Calculate range
67  const T r = thames::vector::geometry::norm3(R);
68 
69  // Precompute factors
70  const T J2_fac1 = -1.5*mu*J2*pow(radius, 2.0)/pow(r, 5.0);
71  const T J2_fac2 = 5.0*pow(z, 2.0)/pow(r, 2.0);
72 
73  // Declare and calculate perturbing acceleration vector
74  const std::vector<T> A = {
75  J2_fac1*x*(1.0 - J2_fac2),
76  J2_fac1*y*(1.0 - J2_fac2),
77  J2_fac1*z*(3.0 - J2_fac2)
78  };
79 
80  // Return perturbing acceleration vector
81  return A;
82  }
