63  void CowellPropagator<T>::derivative(const std::vector<T>& RV, std::vector<T>& RVdot, const T t) const {
64  // Calculate factors
65  const T mu = (m_isNonDimensional) ? m_mu/m_factors->grav : m_mu;
66 
67  // Extract Cartesian state vectors
68  std::vector<T> R = {RV[0], RV[1], RV[2]};
69  std::vector<T> V = {RV[3], RV[4], RV[5]};
70 
71  // Calculate range
72  T r = thames::vector::geometry::norm3(R);
73 
74  // Calculate perturbing acceleration
75  std::vector<T> F = m_perturbation->acceleration_total(t, R, V);
76 
77  // Calculate central body acceleration
78  std::vector<T> G = -mu/pow(r, 3.0)*R;
79 
80  // Calculate acceleration
81  std::vector<T> A = G + F;
82 
83  // Store state derivative
84  for(unsigned int ii=0; ii<3; ii++){
85  RVdot[ii] = V[ii];
86  RVdot[ii+3] = A[ii];
