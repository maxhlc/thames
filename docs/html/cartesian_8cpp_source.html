35  template<class T>
36  void states_to_bounds(const std::vector<std::vector<T>>& states, std::vector<T>& lower, std::vector<T>& upper){
37  // Resize vectors for minimum and maximum values
38  lower.resize(6);
39  upper.resize(6);
40 
41  // Iterate through states
42  for(std::size_t ii=0; ii<states.size(); ii++){
43  // Store values for first state, otherwise compare with current minimum/maximum
44  if(ii == 0){
45  lower = states[ii];
46  upper = states[ii];
47  } else {
48  // Iterate through state variables
49  for(unsigned int jj = 0; jj<6; jj++){
50  // Update minimum if state variable is smaller
51  if(lower[jj] > states[ii][jj])
52  lower[jj] = states[ii][jj];
53 
54  // Update maximum if state variable is larger
55  if(upper[jj] < states[ii][jj])
56  upper[jj] = states[ii][jj];
57  }
58  }
59  }
60  }

63  template<class T>
64  std::vector<T> state_to_sample(const std::vector<T>& state, const std::vector<T>& lower, const std::vector<T>& upper){
65  // Declare vector for sample
66  std::vector<T> sample(state.size());
67 
68  // Iterate through state
69  for(std::size_t ii=0; ii<state.size(); ii++){
70  // Set sample value to zero when the state has zero range, or scale state value between [-1,1] otherwise
71  if((upper[ii] - lower[ii]) == 0.0){
72  sample[ii] = 0.0;
73  } else {
74  sample[ii] = 2.0*(state[ii] - lower[ii])/(upper[ii] - lower[ii]) - 1.0;
75  }
76  }
77 
78  // Return sample point
79  return sample;
80  }
