54  template<class T>
55  T newton_raphson(const std::function<T (T)>& func, const std::function<T (T)>& dfunc, T xn, T tol){
56  // Declare approximation variable
57  T xn1;
58 
59  // Set converged flag to false
60  bool converged = false;
61 
62  // Iterate until converged
63  while(!converged){
64  // Update approximation
65  xn1 = xn - func(xn)/dfunc(xn);
66 
67  // Converged if update is smaller than tolerance
68  if(fabs(xn1 - xn) < tol)
69  converged = true;
70 
71  // Update previous approximation
72  xn = xn1;
73  }
74 
75  // Return root
76  return xn1;
77  }
