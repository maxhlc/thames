72  std::vector<T> BasePropagator<T>::propagate(T tstart, T tend, T tstep, std::vector<T> state, const PropagatorParameters<T> options, const StateTypes statetype) {
73  // Non-dimensionalise
74  if (options.isNonDimensional) {
75  // Update factors
76  // TODO: clean-up
77  m_perturbation->set_nondimensional(true);
78  std::vector<T> state_cartesian = thames::conversions::universal::convert_state<T>(tstart, state, m_mu, statetype, CARTESIAN, m_perturbation);
79  *m_factors = thames::conversions::dimensional::calculate_factors(state_cartesian, m_mu);
80 
81  // Scale times
82  tstart /= m_factors->time;
83  tend /= m_factors->time;
84  tstep /= m_factors->time;
85 
86  // Scale state
87  state = thames::conversions::universal::nondimensionalise_state(state, statetype, *m_factors);
88  }
89 
90  // Calculate gravitational parameters
91  const T mu = (m_isNonDimensional) ? m_mu/m_factors->grav : m_mu;
92 
93  // Set non-dimensional flag
94  m_isNonDimensional = options.isNonDimensional;
95 
96  // Set perturbation non-dimensional flag
97  m_perturbation->set_nondimensional(options.isNonDimensional);
98 
99  // Convert state
100  state = thames::conversions::universal::convert_state<T>(tstart, state, mu, statetype, m_propstatetype, m_perturbation);
101 
102  // Declare state derivative
103  auto func = [this](const std::vector<T>& x, std::vector<T>& dxdt, const T t){return derivative(x, dxdt, t);};
104 
105  // Propagate according to the fixed flag
106  if(options.isFixedStep){
107  // Declare stepper
108  boost::numeric::odeint::runge_kutta4<std::vector<T>> stepper;
109 
110  // Propagate orbit
111  boost::numeric::odeint::integrate_const(stepper, func, state, tstart, tend, tstep);
112  } else {
113  // Declare stepper
114  boost::numeric::odeint::runge_kutta_cash_karp54<std::vector<T>> stepper;
115  auto steppercontrolled = boost::numeric::odeint::make_controlled(options.absoluteTolerance, options.relativeTolerance, stepper);
116 
117  // Propagate orbit
118  boost::numeric::odeint::integrate_adaptive(steppercontrolled, func, state, tstart, tend, tstep);
119  }
120 
121  // Convert state
122  state = thames::conversions::universal::convert_state<T>(tend, state, mu, m_propstatetype, statetype, m_perturbation);
123 
124  // Re-dimensionalise
125  if (options.isNonDimensional) {
126  state = thames::conversions::universal::dimensionalise_state(state, statetype, *m_factors);
127  }
128 
129  // Return final state
130  return state;
131 
132  }
