68  template<class T>
69  std::array<T, 6> BasePropagator<T>::propagate(T tstart, T tend, T tstep, std::array<T, 6> state, const PropagatorOptions<T> options, const StateTypes statetype) {
70  // Check that input is Cartesian state
71  if(statetype != CARTESIAN)
72  throw std::runtime_error("Unsupported state type");
73 
74  // Set non-dimensional flag
75  m_isNonDimensional = options.isNonDimensional;
76 
77  // Set perturbation non-dimensional flag
78  m_perturbation->set_nondimensional(options.isNonDimensional);
79 
80  // Non-dimensionalise
81  if (options.isNonDimensional) {
82  tstart /= m_factors->time;
83  tend /= m_factors->time;
84  tstep /= m_factors->time;
85  state = thames::conversions::dimensional::cartesian_nondimensionalise(state, *m_factors);
86  }
87 
88  // Calculate gravitational parameters
89  const T mu = (m_isNonDimensional) ? m_mu/m_factors->grav : m_mu;
90 
91  // Convert state
92  state = thames::conversions::universal::convert_state(tstart, state, mu, statetype, m_propstatetype, m_perturbation);
93 
94  // Declare state derivative
95  auto func = [this](const std::array<T, 6>& x, std::array<T, 6>& dxdt, const T t){return derivative(x, dxdt, t);};
96 
97  // Propagate according to the fixed flag
98  if(options.isfixedStep){
99  // Declare stepper
100  boost::numeric::odeint::runge_kutta4<std::array<T, 6>> stepper;
101 
102  // Propagate orbit
103  boost::numeric::odeint::integrate_const(stepper, func, state, tstart, tend, tstep);
104  } else {
105  // Declare stepper
106  boost::numeric::odeint::runge_kutta_cash_karp54<std::array<T, 6>> stepper;
107  auto steppercontrolled = boost::numeric::odeint::make_controlled(options.atol, options.rtol, stepper);
108 
109  // Propagate orbit
110  boost::numeric::odeint::integrate_adaptive(steppercontrolled, func, state, tstart, tend, tstep);
111  }
112 
113  // Convert state
114  state = thames::conversions::universal::convert_state(tend, state, mu, m_propstatetype, statetype, m_perturbation);
115 
116  // Re-dimensionalise
117  if (options.isNonDimensional) {
118  state = thames::conversions::dimensional::cartesian_dimensionalise(state, *m_factors);
119  }
120 
121  // Return final state
122  return state;
123 
124  }

229  template<class T, template<class> class P>
230  std::vector<P<T>> BasePropagatorPolynomial<T, P>::propagate(T tstart, T tend, T tstep, std::vector<P<T>> state, PropagatorOptions<T> options, StateTypes statetype) {
231  // Check that input is Cartesian state
232  if(statetype != thames::constants::statetypes::CARTESIAN)
233  throw std::runtime_error("Unsupported state type");
234 
235  // Set non-dimensional flag
236  m_dyn->m_isNonDimensional = options.isNonDimensional;
237 
238  // Set perturbation non-dimensional flag
239  m_perturbation->set_nondimensional(options.isNonDimensional);
240 
241  // Non-dimensionalise
242  if (options.isNonDimensional) {
243  tstart /= m_factors->time;
244  tend /= m_factors->time;
245  tstep /= m_factors->time;
246  state = thames::conversions::dimensional::cartesian_nondimensionalise(state, *m_factors);
247  }
248 
249  // Calculate gravitational parameters
250  const T mu = (options.isNonDimensional) ? m_mu/m_factors->grav : m_mu;
251 
252  // Convert state
253  state = thames::conversions::universal::convert_state(tstart, state, mu, statetype, m_propstatetype, m_perturbation);
254 
255  // Calculate number of steps based on time step
256  unsigned int nstep = (int) ceil((tend - tstart)/tstep);
257 
258  // Create final state vector
259  std::vector<P<T>> statefinal(state);
260 
261  // Propagate according to the fixed flag
262  if(options.isfixedStep){
263  // Create integrator
264  rk4<P<T>> integrator(m_dyn);
265 
266  // Integrate state
267  integrator.integrate(tstart, tend, nstep, state, statefinal);
268  } else {
269  // Create integrator
270  rk45<P<T>> integrator(m_dyn, options.atol, options.rtol);
271 
272  // Integrate state
273  integrator.integrate(tstart, tend, nstep, state, statefinal);
274  }
275 
276  // Convert state
277  statefinal = thames::conversions::universal::convert_state(tend, statefinal, mu, m_propstatetype, statetype, m_perturbation);
278 
279  // Re-dimensionalise
280  if (options.isNonDimensional) {
281  statefinal = thames::conversions::dimensional::cartesian_dimensionalise(statefinal, *m_factors);
282  }
283 
284  // Return final state
285  return statefinal;
286  }
