38

41 
42  // Load JSON
43  nlohmann::json j;

46  // Load metadata

48 
49  // Throw error if not an input file

54  if (j["states"].size() > 1)
55  throw std::runtime_error("Input file must have only one set of points");

58  spacecraft = j["spacecraft"].get<SpacecraftParameters<T>>();

65

68
